def get_age_ratio(age_group):
    if age_group == 'child':
        series = df[df.Age <= 18]['Survived']  
    elif age_group in ('male', 'female'):
        series = df[(df.Sex == age_group) 
                    & (df.Age > 18)]['Survived']
   
        
    saved_ratio = series.sum() / (series.shape[0] * 1.0)
    return saved_ratio
    
    
age_group_saved_df = pd.DataFrame(
